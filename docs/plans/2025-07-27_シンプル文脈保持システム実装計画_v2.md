# 新規作成: src/agri_ai/core/session_manager.py
from redis import Redis
import json
import os
from typing import Dict, Any, Optional

class SessionManager:
    def __init__(self):
        self.redis_client = Redis.from_url(
            os.getenv("REDIS_URL", "redis://localhost:6379")
        )
    
    def _key(self, user_id: str, thread_id: str = "default") -> str:
        return f"session:{user_id}:{thread_id}"
    
    def load_session(self, user_id: str, thread_id: str = "default") -> Dict[str, Any]:
        """セッションデータを読み込み"""
        raw = self.redis_client.get(self._key(user_id, thread_id))
        return json.loads(raw) if raw else {
            "conversation_history": [],
            "pending_confirmation": None
        }
    
    def save_session(self, user_id: str, data: Dict[str, Any], thread_id: str = "default"):
        """セッションデータを保存（24時間TTL）"""
        self.redis_client.setex(
            self._key(user_id, thread_id), 
            86400,  # 24時間
            json.dumps(data, ensure_ascii=False, default=str)
        )
    
    def has_pending_confirmation(self, user_id: str, thread_id: str = "default") -> bool:
        """確認待ち状態かチェック"""
        session = self.load_session(user_id, thread_id)
        return session.get("pending_confirmation") is not None
    
    def get_pending_confirmation(self, user_id: str, thread_id: str = "default") -> Optional[Dict]:
        """確認待ちデータを取得"""
        session = self.load_session(user_id, thread_id)
        return session.get("pending_confirmation")
    
    def clear_pending_confirmation(self, user_id: str, thread_id: str = "default"):
        """確認待ち状態をクリア"""
        session = self.load_session(user_id, thread_id)
        session["pending_confirmation"] = None
        self.save_session(user_id, session, thread_id)
```

# 修正対象: src/agri_ai/core/master_agent.py
from langchain.memory import ConversationBufferMemory
from langchain.memory.chat_message_histories import RedisChatMessageHistory
from .session_manager import SessionManager

class MasterAgent:
    def __init__(self):
        # 既存コード...
        self.session_manager = SessionManager()
    
    def _create_agent_with_memory(self, user_id: str, thread_id: str = "default"):
        """メモリ付きAgentExecutorを作成"""
        
        # RedisChatMessageHistory を使用（LangChain標準）
        message_history = RedisChatMessageHistory(
            session_id=f"{user_id}:{thread_id}",
            url=os.getenv("REDIS_URL", "redis://localhost:6379"),
            ttl=86400  # 24時間
        )
        
        # ConversationBufferMemory にRedis履歴を注入
        memory = ConversationBufferMemory(
            chat_memory=message_history,
            return_messages=True,
            memory_key="chat_history"
        )
        
        # AgentExecutor にメモリを注入
        agent_executor = AgentExecutor(
            agent=self.agent,
            tools=self.tools,
            memory=memory,  # ここが重要！
            verbose=True,
            handle_parsing_errors=True
        )
        
        return agent_executor
```

# 修正対象: src/agri_ai/line_bot/webhook.py
from ..core.session_manager import SessionManager

def handle_message(event):
    user_id = event.source.user_id
    message_text = event.message.text
    
    session_manager = SessionManager()
    
    # 確認待ち状態かチェック
    if session_manager.has_pending_confirmation(user_id):
        return handle_confirmation_response(event, session_manager)
    else:
        return handle_normal_message(event, session_manager)

def handle_confirmation_response(event, session_manager):
    """確認への返答を処理"""
    user_id = event.source.user_id
    message_text = event.message.text
    
    # 「はい」「OK」「了解」などの肯定的な返答をチェック
    if is_affirmative_response(message_text):
        pending_data = session_manager.get_pending_confirmation(user_id)
        
        if pending_data:
            agent_type = pending_data.get("agent_type")
            confirmation_data = pending_data.get("confirmation_data")
            
            # 対応するエージェントの確認処理を呼び出し
            if agent_type == "WorkLogRegistrationAgent":
                result = execute_work_log_confirmation(confirmation_data)
            elif agent_type == "FieldAgent":
                result = execute_field_confirmation(confirmation_data)
            # 他のエージェント...
            
            # 確認待ち状態をクリア
            session_manager.clear_pending_confirmation(user_id)
            
            return create_line_response(result["message"])
    
    # 否定的な返答の場合
    session_manager.clear_pending_confirmation(user_id)
    return create_line_response("処理をキャンセルしました。")

def is_affirmative_response(text: str) -> bool:
    """肯定的な返答かどうかを判定"""
    affirmative_patterns = [
        "はい", "ok", "OK", "オーケー", "了解", "お願いします", 
        "やって", "実行", "登録して", "yes", "Yes", "YES"
    ]
    return any(pattern in text for pattern in affirmative_patterns)

def handle_normal_message(event, session_manager):
    """通常のメッセージ処理"""
    user_id = event.source.user_id
    message_text = event.message.text
    
    # メモリ付きMasterAgentを作成
    master_agent = MasterAgent()
    agent_executor = master_agent._create_agent_with_memory(user_id)
    
    # エージェント実行
    result = agent_executor.invoke({"input": message_text})
    
    # 確認フローが発生した場合はセッションに保存
    if result.get("requires_confirmation"):
        session_manager.save_session(user_id, {
            "pending_confirmation": {
                "agent_type": result.get("agent_type", "Unknown"),
                "confirmation_data": result.get("confirmation_data"),
                "created_at": datetime.now().isoformat()
            }
        })
    
    return create_line_response(result["output"])

        # 既存の登録処理
